
create table id_gen (
    gen_name varchar_ignorecase(80) not null primary key,
    gen_val int
);

create table Recipe( 
	id bigint not null primary key, 
	title varchar_ignorecase(50) not null, 
	contents varchar_ignorecase(50) not null,
	author varchar_ignorecase(50) not null,
	version int not null
);
